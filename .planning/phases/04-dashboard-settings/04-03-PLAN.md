    **Prisma schema additions:**

    model Procedure {
      id             String   @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
      organizationId String   @db.Uuid
      name           String
      duration       Int      // minutes
      price          Decimal? @db.Decimal(10, 2)
      prepInstructions String?
      isActive       Boolean  @default(true)
      createdAt      DateTime @default(now())
      updatedAt      DateTime @updatedAt
      professionals  ProfessionalProcedure[]
      @@unique([organizationId, name])
    }

    model Professional {
      id             String   @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
      organizationId String   @db.Uuid
      name           String
      email          String?
      specialties    String[] // array of strings
      workingHoursStart Int   @default(8)
      workingHoursEnd   Int   @default(18)
      workingDays    Int[]    @default([1,2,3,4,5]) // 0=Sun, 1=Mon...
      calendarTokenId String? @db.Uuid
      isActive       Boolean  @default(true)
      createdAt      DateTime @default(now())
      updatedAt      DateTime @updatedAt
      procedures     ProfessionalProcedure[]
      calendarToken  CalendarToken? @relation(fields: [calendarTokenId], references: [id])
    }

    model ProfessionalProcedure {
      professionalId String      @db.Uuid
      procedureId    String      @db.Uuid
      professional   Professional @relation(fields: [professionalId], references: [id], onDelete: Cascade)
      procedure      Procedure    @relation(fields: [procedureId], references: [id], onDelete: Cascade)
      @@id([professionalId, procedureId])
    }

    model MessageTemplate {
      id             String   @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
      organizationId String   @db.Uuid
      type           String   // 'followup' | 'notification' | 'confirmation' | 'reminder' | 'no_show'
      name           String
      content        String   // Template with {{variable}} placeholders
      isActive       Boolean  @default(true)
      createdAt      DateTime @default(now())
      updatedAt      DateTime @updatedAt
      @@unique([organizationId, name])
    }
